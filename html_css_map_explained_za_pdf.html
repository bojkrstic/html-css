Responsive 
Responsive ponašanje je u CSS-u, ne u HTML-u.
 Media queries 
@media (max-width: 960px) { /* Tablet i uži ekran */
  .sadrzaj { /* Glavni raspored */
    flex-direction: column; /* Kolone idu jedna ispod druge */
  }

  .desna-strana { /* Sidebar ispod levog dela */
    flex-basis: auto; /* Uklanja fiksno ponašanje */
    min-width: 0; /* Dozvoljava skupljanje */
  }

  .kurs { /* Kartice kurseva na užem ekranu */
    width: 100%; /* Puna širina */
    margin-right: 0; /* Bez desnog razmaka */
  }
}

@media (max-width: 640px) { /* Mali ekran */
  .strana { /* Glavni wrapper */
    padding: 12px; /* Manji padding */
  }

  nav { /* Navigacija */
    flex-direction: column; /* Vertikalno slaganje */
    align-items: flex-start; /* Levo poravnanje */
  }

  .baner-naslov { /* Naslov bannera */
    padding-top: 70px; /* Manje spuštanje */
    font-size: 24px; /* Manji font */
  }
}
